import java.util.Scanner; public class Solution { public static void main(String[] args) { Scanner sc = new Scanner(System.in); int n = sc.nextInt(); int[] freq = new int[100]; for (int i = 0; i < n; i++) { int num = sc.nextInt(); freq[num]++; } for (int i = 0; i < 100; i++) { System.out.print(freq[i] + " "); } } }
